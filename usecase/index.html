Docker Compose

22% less code
YAML (Before)
services:
  web:
    build: .
    ports:
      - "3000:3000"
    depends_on:
      - api
      - db
    environment:
      NODE_ENV: production
      API_URL: http://api:8080

  db:
    image: postgres:15-alpine
    environment:
      POSTGRES_DB: myapp
    volumes:
      - pgdata:/var/lib/postgresql/data

volumes:
  pgdata:
GOML (After)
services {
  web {
    build = .
    ports = [3000:3000]
    depends_on = [api, db]
    environment {
      NODE_ENV = production
      API_URL = http://api:8080
    }
  }

  db {
    image = postgres:15-alpine
    environment {
      POSTGRES_DB = myapp
    }
    volumes = [pgdata:/var/lib/postgresql/data]
  }
}

volumes { pgdata }
35 lines vs 45 in YAML
0 indentation errors

GitHub Actions

14% less code
YAML (Before)
name: CI/CD Pipeline

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  test:
    name: Run Tests
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-go@v5
        with:
          go-version: '1.21'
      - run: go test ./... -v
GOML (After)
pipeline {
  name = CI/CD Pipeline

  on {
    push = [main, develop]
    pull_request = [main]
  }

  jobs {
    test {
      name = Run Tests
      runs_on = ubuntu-latest
      steps [
        { uses = actions/checkout@v4 }
        { uses = actions/setup-go@v5
          with { go_version = "1.21" } }
        { run = "go test ./... -v" }
      ]
    }
  }
}
30 lines vs 35 in YAML
Shallow nesting depth

Kubernetes

17% less code
YAML (Before)
apiVersion: apps/v1
kind: Deployment
metadata:
  name: api-server
  namespace: production
  labels:
    app: api
spec:
  replicas: 3
  selector:
    matchLabels:
      app: api
  template:
    metadata:
      labels:
        app: api
    spec:
      containers:
        - name: api
          image: ghcr.io/myorg/api:latest
          ports:
            - containerPort: 8080
GOML (After)
deployment {
  apiVersion = apps/v1
  kind = Deployment

  metadata {
    name = api-server
    namespace = production
    labels { app = api }
  }

  spec {
    replicas = 3
    selector { match_labels { app = api } }

    template {
      metadata { labels { app = api } }

      spec {
        containers [{
          name = api
          image = ghcr.io/myorg/api:latest
          ports = [{ container_port = 8080 }]
        }]
      }
    }
  }
}
25 lines vs 30 in YAML
Explicit structure
